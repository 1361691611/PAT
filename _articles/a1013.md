## 思路

题目即要求我们计算这些城市有几个内部相连的部分。

思路是对这些城市和公路组成的图进行多次遍历，直至所有城市都被遍历位置，遍历次数减1即为需要修复的公路数量。

具体遍历时，逐一城市进行一个更高层的循环，跳过已遍历过的即可。

{% raw %}#include <stdio.h>

#define MAX 1000

void DFS(int v, int known[], int G[][MAX])
{
    known[v] = 1;
    for(int i = 0; i < MAX; i++)
        if((G[v][i] || G[i][v]) && !known[i])
            DFS(i, known, G);
}

int main()
{
    int N, M, K;
    scanf("%d %d %d", &N, &M, &K);

    int Graph[MAX][MAX] = {{0}};

    int city1, city2;
    for(int i = 0; i < M; i++)
    {
        scanf("%d %d", &city1, &city2);
        Graph[city1][city2] = 1;
    }

    for(int i = 0; i < K; i++)
    {
        int known[MAX] = {0}, lostcity, count = 0;
        scanf("%d", &lostcity);

        known[lostcity] = 1;
        for(int i = 1; i <= N; i++) if(!known[i])
        {
            DFS(i, known, Graph);
            count++;
        }
        printf("%d\n", count - 1);
    }
}
